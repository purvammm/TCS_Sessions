// 2. Connection URL (file-based, auto-creates):
jdbc:h2:~/employeedb

// 3. Maven dependency:
<dependency>
    <groupId>com.h2database</groupId>
    <artifactId>h2</artifactId>
    <version>2.2.224</version>

package com.tcs.jdbc;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.SQLException;

public class JDBCTest {
    public static void main(String[] args) {
        try {
            // Step 1: Register the driver (optional in JDBC 4.0+, shown for clarity)
            Class.forName("org.apache.derby.jdbc.EmbeddedDriver");

            // Step 2: Establish the connection
            String url = "jdbc:derby:employeedb;create=true";
            Connection con = DriverManager.getConnection(url, "sa", "");

            // Step 3-4: Verify the connection is open
            System.out.println("Connected: " + !con.isClosed());

            // Step 5: Close it
            con.close();
            System.out.println("Closed: " + con.isClosed());

        } catch (ClassNotFoundException e) {
            System.out.println("Driver not found: " + e.getMessage());
        } catch (SQLException e) {
            System.out.println("DB error: " + e.getMessage());
        }
    }
}

package com.tcs.jdbc;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.SQLException;
import java.sql.Statement;

public class CreateAndInsert {

    public static void main(String[] args) {
        String url = "jdbc:derby:employeedb;create=true";

        try (Connection con = DriverManager.getConnection(url, "sa", "")) {

            // ---- STEP 1: create the table (only succeeds the first time) ----
            createTable(con);

            // ---- STEP 2: insert 3 employees ----
            insertEmployee(con, 101, "Ravi",  "Sharma", "Engineering", 75000.00);
            insertEmployee(con, 102, "Priya", "Mehta",  "HR",          65000.00);
            insertEmployee(con, 103, "Arjun", "Patel",  "Engineering", 80000.00);

        } catch (SQLException e) {
            System.out.println("DB error: " + e.getMessage());
        }
    }

    // Creates the employees table. If it already exists, we just print a note
    // instead of crashing — so you can re-run this file safely.
    public static void createTable(Connection con) {
        String sql = "CREATE TABLE employees (" +
                     "emp_id INT PRIMARY KEY, " +
                     "first_name VARCHAR(50), " +
                     "last_name VARCHAR(50), " +
                     "department VARCHAR(50), " +
                     "salary DOUBLE)";
        try (Statement stmt = con.createStatement()) {
            stmt.executeUpdate(sql);
            System.out.println("Table created successfully.");
        } catch (SQLException e) {
            System.out.println("Table already exists — skipping create. (" + e.getSQLState() + ")");
        }
    }

    // Inserts one employee using a PreparedStatement (the safe way).
    public static void insertEmployee(Connection con, int id, String first,
                                      String last, String dept, double salary)
                                      throws SQLException {
        String sql = "INSERT INTO employees VALUES (?, ?, ?, ?, ?)";
        try (PreparedStatement ps = con.prepareStatement(sql)) {
            ps.setInt(1, id);          // parameter numbers start at 1, not 0
            ps.setString(2, first);
            ps.setString(3, last);
            ps.setString(4, dept);
            ps.setDouble(5, salary);
            int rows = ps.executeUpdate();
            System.out.println(rows + " row(s) inserted.");
        }
    }
}
Expected output (first run)
Table created successfully. 1 row(s) inserted. 1 row(s) inserted. 1 row(s) inserted.
On the second run you'll instead see Table already exists — skipping create. at the top, then the duplicate-key errors (next box) — that's expected.
Understand what you pasted Part A is the createTable method — it runs the SQL CREATE TABLE using a plain Statement. Part B is the insertEmployee method — it runs INSERT using a PreparedStatement with ? placeholders. The main method at the top calls them in order: create first, then insert three times.
Verify it really worked In Data Source Explorer: refresh your connection → expand Tables → EMPLOYEES now appears. Right-click EMPLOYEES → Data → Edit to see all 3 rows.
Observation — try breaking it on purpose Run the file a second time. The same 3 IDs already exist, so each insert now fails with a duplicate-primary-key error. Read it carefully and note the SQLState code — it's 23505. We'll handle this gracefully in Lab 5. To start clean, delete the rows in Data Source Explorer (or pick new IDs like 201/202/203).

package com.tcs.jdbc;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.ResultSet;
import java.sql.SQLException;
import java.sql.Statement;

public class SelectAll {

    public static void main(String[] args) {
        String url = "jdbc:derby:employeedb;create=true";
        String sql = "SELECT * FROM employees";

        try (Connection con = DriverManager.getConnection(url, "sa", "");
             Statement stmt = con.createStatement();
             ResultSet rs = stmt.executeQuery(sql)) {

            System.out.println("ID  | Name            | Department  | Salary");
            System.out.println("----+-----------------+-------------+--------");

            while (rs.next()) {   // moves to the next row; false when no rows left
                System.out.printf("%-4d| %-16s| %-12s| %.2f%n",
                    rs.getInt("emp_id"),
                    rs.getString("first_name") + " " + rs.getString("last_name"),
                    rs.getString("department"),
                    rs.getDouble("salary"));
            }

        } catch (SQLException e) {
            System.out.println("DB error: " + e.getMessage());
        }
    }
}
Expected output
ID | Name | Department | Salary ----+-----------------+-------------+-------- 101 | Ravi Sharma | Engineering | 75000.00 102 | Priya Mehta | HR | 65000.00 103 | Arjun Patel | Engineering | 80000.00
How the cursor works rs.next() starts before row 1. Each call advances one row and returns true while rows remain, false when they're exhausted — so the while loop ends naturally. You read columns by name (rs.getString("first_name")) or by 1-based index.
Try it yourself Add a method getEmployeesByDepartment(String dept) that returns only Engineering employees. Hint: use WHERE department = ? with a PreparedStatement and bind the value with ps.setString(1, dept).

📄UpdateDeleteSearch.javapaste the WHOLE file
package com.tcs.jdbc;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.sql.SQLException;

public class UpdateDeleteSearch {

    static final String URL = "jdbc:derby:employeedb;create=true";

    public static void main(String[] args) {
        searchById(101);   // exists  -> "Found: ..."
        searchById(999);   // missing -> "No employee found..."

        updateSalary(101, 90000.00);
        deleteById(103);

        System.out.println("\n--- Table after update & delete ---");
        printAll();        // confirm the changes landed
    }

    // ===== SELECT BY ID — use if(rs.next()), not a while loop =====
    public static void searchById(int empId) {
        String sql = "SELECT * FROM employees WHERE emp_id = ?";
        try (Connection con = DriverManager.getConnection(URL, "sa", "");
             PreparedStatement ps = con.prepareStatement(sql)) {
            ps.setInt(1, empId);
            try (ResultSet rs = ps.executeQuery()) {
                if (rs.next()) {
                    System.out.println("Found: " + rs.getString("first_name")
                        + " " + rs.getString("last_name"));
                } else {
                    System.out.println("No employee found with ID: " + empId);
                }
            }
        } catch (SQLException e) {
            System.out.println("Search failed: " + e.getMessage());
        }
    }

    // ===== UPDATE =====
    public static void updateSalary(int empId, double newSalary) {
        String sql = "UPDATE employees SET salary = ? WHERE emp_id = ?";
        try (Connection con = DriverManager.getConnection(URL, "sa", "");
             PreparedStatement ps = con.prepareStatement(sql)) {
            ps.setDouble(1, newSalary);
            ps.setInt(2, empId);
            int rows = ps.executeUpdate();
            System.out.println(rows > 0 ? "Salary updated." : "Employee not found.");
        } catch (SQLException e) {
            System.out.println("Update failed: " + e.getMessage());
        }
    }

    // ===== DELETE =====
    public static void deleteById(int empId) {
        String sql = "DELETE FROM employees WHERE emp_id = ?";
        try (Connection con = DriverManager.getConnection(URL, "sa", "");
             PreparedStatement ps = con.prepareStatement(sql)) {
            ps.setInt(1, empId);
            int rows = ps.executeUpdate();
            System.out.println(rows > 0 ? "Deleted." : "Employee not found.");
        } catch (SQLException e) {
            System.out.println("Delete failed: " + e.getMessage());
        }
    }

    // ===== helper: print everything so you can SEE the changes =====
    public static void printAll() {
        String sql = "SELECT * FROM employees";
        try (Connection con = DriverManager.getConnection(URL, "sa", "");
             PreparedStatement ps = con.prepareStatement(sql);
             ResultSet rs = ps.executeQuery()) {
            while (rs.next()) {
                System.out.printf("%-4d| %-16s| %-12s| %.2f%n",
                    rs.getInt("emp_id"),
                    rs.getString("first_name") + " " + rs.getString("last_name"),
                    rs.getString("department"),
                    rs.getDouble("salary"));
            }
        } catch (SQLException e) {
            System.out.println("Read failed: " + e.getMessage());
        }
    }
}
Expected output
Found: Ravi Sharma No employee found with ID: 999 Salary updated. Deleted. --- Table after update & delete --- 101 | Ravi Sharma | Engineering | 90000.00 102 | Priya Mehta | HR | 65000.00
Notice 101's salary is now 90000.00 and row 103 is gone.
Why if(rs.next()) for search? The id is the primary key, so a search returns at most one row. A single if(rs.next()) is clearer than a loop and gives you a clean “not found” branch.
Why the rows-affected check matters executeUpdate() returns the number of rows changed. Checking rows > 0 lets you tell “query ran fine but matched nothing” apart from “record found and modified” — far more useful than assuming success.
LAB 5▶ try-with-resources Refactor
The problem we're fixing Every lab so far has a risk: if an exception fires between getConnection() and con.close(), the connection leaks. try-with-resources auto-closes every resource — on success and on failure — eliminating the leak. package com.tcs.jdbc;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.SQLException;

public class DBConnector {
    private static final String URL = "jdbc:derby:employeedb;create=true";
    private static final String USER = "sa";
    private static final String PASSWORD = "";

    public static Connection getConnection() throws SQLException {
        return DriverManager.getConnection(URL, USER, PASSWORD);
    }
}

package com.tcs.jdbc;

import java.sql.Connection;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.sql.SQLException;

public class EmployeeDAO {

    // ===== CREATE =====
    public void insert(int id, String first, String last,
                       String dept, double salary) {
        String sql = "INSERT INTO employees VALUES (?, ?, ?, ?, ?)";
        try (Connection con = DBConnector.getConnection();
             PreparedStatement ps = con.prepareStatement(sql)) {
            ps.setInt(1, id);
            ps.setString(2, first);
            ps.setString(3, last);
            ps.setString(4, dept);
            ps.setDouble(5, salary);
            int rows = ps.executeUpdate();
            System.out.println(rows + " row(s) inserted.");
        } catch (SQLException e) {
            System.out.println("Insert failed [" + e.getSQLState() + "]: "
                               + e.getMessage());
        }
    }

    // ===== READ ALL =====
    public void getAll() {
        String sql = "SELECT * FROM employees";
        try (Connection con = DBConnector.getConnection();
             PreparedStatement ps = con.prepareStatement(sql);
             ResultSet rs = ps.executeQuery()) {
            System.out.println("ID  | Name            | Department  | Salary");
            System.out.println("----+-----------------+-------------+--------");
            while (rs.next()) {
                System.out.printf("%-4d| %-16s| %-12s| %.2f%n",
                    rs.getInt("emp_id"),
                    rs.getString("first_name") + " " + rs.getString("last_name"),
                    rs.getString("department"),
                    rs.getDouble("salary"));
            }
        } catch (SQLException e) {
            System.out.println("Read failed: " + e.getMessage());
        }
    }

    // ===== READ BY ID =====
    public void getById(int id) {
        String sql = "SELECT * FROM employees WHERE emp_id = ?";
        try (Connection con = DBConnector.getConnection();
             PreparedStatement ps = con.prepareStatement(sql)) {
            ps.setInt(1, id);
            try (ResultSet rs = ps.executeQuery()) {
                if (rs.next()) {
                    System.out.println("Found: " + rs.getString("first_name")
                        + " " + rs.getString("last_name")
                        + " (" + rs.getString("department") + ")");
                } else {
                    System.out.println("No employee found with ID: " + id);
                }
            }
        } catch (SQLException e) {
            System.out.println("Read failed: " + e.getMessage());
        }
    }

    // ===== UPDATE =====
    public void updateSalary(int id, double salary) {
        String sql = "UPDATE employees SET salary = ? WHERE emp_id = ?";
        try (Connection con = DBConnector.getConnection();
             PreparedStatement ps = con.prepareStatement(sql)) {
            ps.setDouble(1, salary);
            ps.setInt(2, id);
            int rows = ps.executeUpdate();
            System.out.println(rows > 0 ? "Salary updated." : "Employee not found.");
        } catch (SQLException e) {
            System.out.println("Update failed: " + e.getMessage());
        }
    }

    // ===== DELETE =====
    public void delete(int id) {
        String sql = "DELETE FROM employees WHERE emp_id = ?";
        try (Connection con = DBConnector.getConnection();
             PreparedStatement ps = con.prepareStatement(sql)) {
            ps.setInt(1, id);
            int rows = ps.executeUpdate();
            System.out.println(rows > 0 ? "Deleted." : "Employee not found.");
        } catch (SQLException e) {
            System.out.println("Delete failed: " + e.getMessage());
        }
    }
}

package com.tcs.jdbc;

import java.util.Scanner;

public class EmployeeTester {
    public static void main(String[] args) {
        EmployeeDAO dao = new EmployeeDAO();
        Scanner sc = new Scanner(System.in);

        while (true) {
            System.out.println("\n==== Employee Menu ====");
            System.out.println("1 - Add Employee");
            System.out.println("2 - View All Employees");
            System.out.println("3 - Search by ID");
            System.out.println("4 - Update Salary");
            System.out.println("5 - Delete Employee");
            System.out.println("0 - Exit");
            System.out.print("Choose: ");
            int choice = sc.nextInt();

            switch (choice) {
                case 1:
                    System.out.print("ID: ");        int id = sc.nextInt();
                    System.out.print("First name: "); String f = sc.next();
                    System.out.print("Last name: ");  String l = sc.next();
                    System.out.print("Department: "); String d = sc.next();
                    System.out.print("Salary: ");      double s = sc.nextDouble();
                    dao.insert(id, f, l, d, s);
                    break;
                case 2:
                    dao.getAll();
                    break;
                case 3:
                    System.out.print("ID to search: ");
                    dao.getById(sc.nextInt());
                    break;
                case 4:
                    System.out.print("ID: ");          int uid = sc.nextInt();
                    System.out.print("New salary: ");  double ns = sc.nextDouble();
                    dao.updateSalary(uid, ns);
                    break;
                case 5:
                    System.out.print("ID to delete: ");
                    dao.delete(sc.nextInt());
                    break;
                case 0:
                    System.out.println("Goodbye!");
                    sc.close();
                    return;
                default:
                    System.out.println("Invalid choice. Try again.");
            }
        }
    }
}

Quick reference card

Interfaces & classes

Interface / ClassPurpose
DriverManagerManages drivers, returns a Connection from a URL
ConnectionOpen DB session — one socket, always close it
StatementStatic SQL — never use for user input
PreparedStatementParameterized SQL — always use this
ResultSetCursor over SELECT results, iterate with rs.next()
SQLExceptionChecked exception — getMessage, getSQLState, getErrorCode

The five steps

  1. Class.forName(driverClass) — register (auto in JDBC 4.0+)
  2. DriverManager.getConnection(url, user, pass) — connect
  3. con.prepareStatement(sql) — prepare
  4. ps.executeQuery() or ps.executeUpdate() — execute
  5. con.close() — release (or use try-with-resources)
